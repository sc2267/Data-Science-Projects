print('Split 1')
X_train, X_test, y_train, y_test = train_test_split(features2, buildyear, test_size = 0.20, random_state = 61)
knn = KNeighborsClassifier(n_neighbors = 3)
knn.fit(X_train, y_train) 
predict = knn.predict(X_test)


actual = np.zeros(len(X_test))
prediction = np.zeros(len(X_test))

zeroes = 0
ones = 0
y_test.index=range(len(X_test))
for x in range(len(X_test)):
    if(y_test[x]< 1970):
        actual[x] = 0
        zeroes += 1
    elif(y_test[x] >= 1970):
        actual[x] = 1
        ones += 1
        
baseline = 0

if(ones >= zeroes):
    baseline = ones/(ones+zeroes)
else:
    baseline = zeroes/(ones+zeroes)
    

for x in range(len(X_test)):
    if(predict[x]< 1970):
        prediction[x] = 0

    elif(predict[x] >= 1970):
        prediction[x] = 1

        
predictioncount = 0
for x in range(len(X_test)):
    if(prediction[x]==actual[x]):
        predictioncount += 1
        
predictionvalue= predictioncount/(ones+zeroes)
accuracy = (predictionvalue-baseline)/baseline

print("baseline: " + str(baseline))
print("accuracy: " + str(accuracy))
